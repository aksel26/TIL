# 자릿수 합

### 내 코드

```c++
#include <stdio.h>

using namespace std;
int sum = 0;
int max = -10000;
int digit_sum(int x)
{

    for (int i = x; i > 0; i = i / 10)
    {
        sum = sum + i % 10;
    }

    if (max < sum)
        max = sum;
    
    return sum;
}
int main()
{
    int n;
    scanf("%d", &n);
    int nums;

    for (int i = 0; i < n; i++)
    {
        scanf("%d", &nums);
    }
    printf("%d",digit_sum(nums));

    return 0;
}
```

- `digit_sum ` 함수 안에서 최댓값을 바꾸려고 햇다. 
  - <u>함수 이름을 잘 기억해야 할 것</u>
- 변수를 하나 더 사용했으면 자릿수의 최댓값을 가지는 입력 숫자를 출력할 수 있었다.

### 참고

```c
#include <stdio.h>

using namespace std;

int max = -10000;

// 자릿수의 합을 구하는 함수
int digit_sum(int x)
{
    int sum = 0, temp;
  
    while (x > 0)
    {
        temp = x % 10;
        sum += temp;
        x = x / 10;
    }

    return sum;
}
int main()
{
    int n, sum, res;
    scanf("%d", &n);
    int nums;

    for (int i = 0; i < n; i++)
    {
        scanf("%d", &nums);
        sum = digit_sum(nums);
      
      // 최댓값
        if (sum > max)
        {
            max = sum;
            res = nums;
        }
      
      // 최댓값이 같다면 큰 값을 결과 값으로 바꾼다.
        else if (sum == max)
        {
            if (nums > res)
                res = nums;
        }
    }

    printf("%d\n", res);

    return 0;
}
```
