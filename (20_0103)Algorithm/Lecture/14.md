# 뒤집은 소수

### 내 코드

```c
#include <stdio.h>
#include <string>
#include <math.h>

bool isPrime(int x)
{
    if (x == 1)
        return false;
  
  // 입력받은 수의 제곱근만 연산하기 위해
    int sqrtNum = (int)sqrt(x);
  
    for (int i = 2; i <= sqrtNum; i++)
    {
        if (x % i == 0)
        {
            return false;
        }
    }
    return true;
}

int reverse(int x)
{
    char a[100001];

  // int to char로 변환해서 배열에 넣기
    sprintf(a, "%d", x);

  // 오른쪽 인덱스
    int right = strlen(a) - 1;
  // 왼쪽 인덱스
    int left = 0;

    while (1)
    {
      // 자리 바꿈
        int temp = a[right];
        a[right] = a[left];
        a[left] = temp;
			// 왼쪽 인덱스 중앙으로
        left++;
      // 오른쪽 인덱스 중앙으로
        right--;
			// 인덱스가 같은 경우(홀수 길이) 혹은 왼쪽이 오른쪽을 넘어서는 경우(짝수 길이) break;
        if (left == right || left > right)
            break;
    }
	// 다시 char -> int형으로 변환
    int res = atoi(a);

    return res;
}

int main()
{
    int n, tmp;
    scanf("%d", &n);
    int nums;
    for (int i = 0; i < n; i++)
    {
        scanf("%d", &nums);
        int tmp = reverse(nums);
        if (isPrime(tmp))
        {
            printf("%d ", tmp);
        }
    }

    return 0;
}

```

### Solution

```c
#include <stdio.h>
#include <string>
#include <math.h>

bool isPrime(int x)
{
    if (x == 1)
        return false;

    int sqrtNum = (int)sqrt(x);

    for (int i = 2; i <= sqrtNum; i++)
    {

        if (x % i == 0)
        {
            return false;
            break;
        }
    }

    return true;
}

int reverse(int x)
{

    int res = 0;

    while (x > 0)
    {
        int tmp = x % 10;
        res = res * 10 + tmp;
        x = x / 10;
    }

    return res;
}

int main()
{
    int n, tmp;
    scanf("%d", &n);
    int nums;
    for (int i = 0; i < n; i++)
    {
        scanf("%d", &nums);
        tmp = reverse(nums);
        if (isPrime(tmp))
        {
            printf("%d ", tmp);
        }
    }

    return 0;
}

```
