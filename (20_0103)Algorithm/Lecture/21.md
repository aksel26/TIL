# 카드게임

### 내 코드

```c
#include <stdio.h>
#include <time.h>

int main()
{
    clock_t start, end;
    start = clock();
    double clockResult;

    int a[11];
    int b[11];
    int scoreA = 0;
    int scoreB = 0;
    for (int i = 0; i < 10; i++)
    {
        scanf("%d", &a[i]);
    }
    for (int i = 0; i < 10; i++)
    {
        scanf("%d", &b[i]);
    }

    for (int i = 0; i < 10; i++)
    {
        if (a[i] > b[i])
        {
            scoreA += 3;
        }
        else if (a[i] < b[i])
        {
            scoreB += 3;
        }
        else if (a[i] == b[i])
        {
            scoreA += 1;
            scoreB += 1;
        }
    }

    printf("%d %d\n", scoreA, scoreB);

    if (scoreA > scoreB)
    {
        printf("A\n");
    }
    else if (scoreA == scoreB)
    {

        for (int i = 9; i >= 0; i--)
        {
            if (a[i] == b[i])
            {
                continue;
            }
            else if (a[i] > b[i])
            {
                printf("A\n");
                break;
            }
            else if (a[i] < b[i])
            {
                printf("B\n");
                break;
            }
        }
    }
    else
        printf("B\n");

    end = clock(); //시간 측정 끝
    clockResult = ((double)(end - start) / CLOCKS_PER_SEC);
    printf("%f", clockResult);

    return 0;
}
```

- 마지막 라운드에서 동점이 날 경우 가장 마지막의 승자가 게임의 승자
  - 이부분을 처리하는 과정에서 for문을 반대쪽 방향으로 돌려 출력함
  - 밑에 처럼 for문을 돌지 않더라도 lw변수를 하나 추가하면 결과를 출력할 수 있다.

### Solution

```c
#include <stdio.h>
#include <time.h>

int main()
{
    clock_t start, end;
    start = clock();
    double clockResult;

    int a[11];
    int b[11];
    int scoreA = 0;
    int scoreB = 0;
    int lw = 0;
    for (int i = 0; i < 10; i++)
    {
        scanf("%d", &a[i]);
    }
    for (int i = 0; i < 10; i++)
    {
        scanf("%d", &b[i]);
    }

    for (int i = 0; i < 10; i++)
    {
        if (a[i] > b[i])
        {
            scoreA += 3;
            lw = 1;
        }
        else if (a[i] < b[i])
        {
            scoreB += 3;
            lw = 2;
        }
        else if (a[i] == b[i])
        {
            scoreA += 1;
            scoreB += 1;
        }
    }

    printf("%d %d\n", scoreA, scoreB);

    if (scoreA > scoreB)
    {
        printf("A\n");
    }
    else if (scoreA < scoreB)
    {
        printf("B\n");
    }
    else if (scoreA == scoreB)
    {
        if (lw == 1)
            printf("A\n");
        else if (lw == 2)
            printf("B\n");
        else if (lw == 0)
            printf("D\n");
    }

    end = clock(); //시간 측정 끝
    clockResult = ((double)(end - start) / CLOCKS_PER_SEC);
    printf("%f", clockResult);

    return 0;
}
```
