# 연속 부분 증가수열

### 내 코드

```c
#include <stdio.h>

int main()
{

    int n;
    scanf("%d", &n);
    int a[n];
    for (int i = 0; i < n; i++)
    {
        scanf("%d", &a[i]);
    }
    int cnt = 1;
    int max = -2147000;
    for (int i = 0; i < n; i++)
    {
        if (a[i] <= a[i + 1])
        {
            cnt++;
            if (cnt > max)
            {
                max = cnt;
            }
        }
        else if (a[i] > a[i + 1])
        {
            cnt = 1;
        }
    }

    printf("%d\n", max);

    return 0;
}
```

- 배열을 먼저 다 받고 배열 인덱스를 진행하며 비교한다.
- 비교 할때마다 최댓값을 갱신하는 방식

### Solution

```c
#include <stdio.h>

int main()
{

    int n, i, pre, now;

    scanf("%d", &n);

    scanf("%d", &pre);
    int cnt = 1;
    int max = 1;
    for (i = 2; i <= n; i++)
    {
        scanf("%d", &now);
        if (now >= pre)
        {
            cnt++;
            if (cnt > max)
            {
                max = cnt;
            }
        }
        else
        {
            cnt = 1;
        }
        pre = now;
    }

    printf("%d\n", max);
    return 0;
}
```

- 입력을 받을때마다 비교하여 최댓값을 갱신하는 방식
