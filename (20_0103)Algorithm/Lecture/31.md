# 탄화수소 질량

### 내 코드

```c
#include <stdio.h>

int main()
{

    char a[10];
    int i;
    int resC = 0;
    int resH = 0;
    int total = 0;
    scanf("%s", &a);

    for (i = 0; a[i] != '\0'; i++)
    {
        if (a[i] == 'C')
        {
            resC += 12;
            if (a[i + 1] >= 49 && a[i + 1] <= 57)
            {
                resC = resC * (a[i + 1] - 48);
            }
        }
        else if (a[i] == 'H')
        {
            resH += 1;
            if (a[i + 1] >= 49 && a[i + 1] <= 57)
            {
                resH = resH * (a[i + 1] - 48);
            }
        }
    }

    total = resH + resC;

    printf("%d\n", total);
    return 0;
}
```

- 정답은 나왔지만, 
- 알파벳 숫자 뒤에 1의자리 수인 경우만 고려함

---

<br/> 

### Solution 

```c
#include <stdio.h>

int main()
{
    char a[10];

    int c = 0, h = 0, pos, i;

    scanf("%s", &a);

    if (a[1] == 'H')
    {
        c = 1;
        pos = 1;
    }
    else
    {
        for (i = 1; a[i] != 'H'; i++)
        {
            c = c * 10 + (a[i] - 48);
        }
        pos = i;
        printf("%d\n", pos);
    }
    if (a[pos + 1] == '\0')
    {
        h = 1;
    }
    else
    {
        for (i = pos + 1; a[i] != '\0'; i++)
        {
            h = h * 10 + (a[i] - 48);
        }
    }

    printf("%d\n", c * 12 + h);

    return 0;
}
```

- 첫 단어가 H 이면  `c=1` 을 하고, 포지션(`pos = 1`) 을 이동시킨다.
- 이후 다음 알파벳이 나오기 전까지 숫자화를 시켜준다.
